struct LikeButton: View {
    
    @State private var isLiked: Bool = false
   
    var body: some View {
        Button {
            // 2. The Action: Toggle the boolean (true becomes false, false becomes true)
            isLiked.toggle()
        } label: {
            // 3. The Label: Use logic to change the appearance
            HStack {
                Image(systemName: isLiked ? "heart.fill" : "heart")
                Text(isLiked ? "Liked" : "Like")
            }
            .font(.title)
            .padding()
            // 4. Change color based on state
            .foregroundColor(isLiked ? .white : .blue)
            .background(isLiked ? Color.red : Color.blue.opacity(0.1))
            .cornerRadius(10)
        }
       
    }
}
